GMaps.js

Google Maps API with less pain and more fun.

Basic Map Example

Example:
JAVASCRIPT Code: 
// Basic Map Example
if (typeof GMaps === 'undefined') {
	return;
}
if ($('#basic-map').exists()) {
	basicMapDemo = new GMaps({
	  div: '#basic-map',
	  lat: -36.8270742,
	  lng: -73.0502464,
	  zoom: 13
	});
	
}
